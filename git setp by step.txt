# 1. Navigate to your project folder
cd /path/to/your/project

# 2. Initialize Git (only once per project)
git init

# 3. (Only needed once on your system – sets your identity for all projects)
git config --global user.name "Your Full Name"
git config --global user.email "your.email@example.com"

# 4. (Only needed once per project) – Link your GitHub repo
git remote add origin https://github.com/YOUR-USERNAME/YOUR-REPO-NAME.git

# 5. Add all files to be tracked (or specify just one file, e.g., README.md)
git add .

# 6. Commit the added files with a message
git commit -m "Initial commit / Updated README / etc."

# 7. Push to GitHub (for first push)
git push -u origin main
